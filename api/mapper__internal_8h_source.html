641 inline static int mapper_type_size(char type)
642 {
643  switch (type) {
644  case 'i': return sizeof(int);
645  case 'b':
646  case 'T':
647  case 'F': return sizeof(int);
648  case 'f': return sizeof(float);

654  case 'c': return sizeof(char);

656  die_unless(0, "Unknown type '%c' in mapper_type_size().\n", type);
