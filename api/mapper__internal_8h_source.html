823 {
824  mapper_signal_value_t v;
825  switch (type) {
826  case 'i': return sizeof(v.i32);
827  case 'f': return sizeof(v.f);
828  case 'd': return sizeof(v.d);
829  default:
830  die_unless(0, "getting size of unknown type %c\n", type);
831  return 0;
832  }
